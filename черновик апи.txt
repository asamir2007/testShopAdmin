
shopAdmin.php - страница для админа
	
	следующая страница товаров(список) для таблицы
		/items
		HTTP REQUEST / METHOD=GET
			m=GET
			sizePage//размер страницы
			idElement//ид элемента после которого расположен 1й элемент страницы.  (это самый быстрый метод запроса из БД и позволяет получать верные данные даже если в процессе листания были добавлены или удалены элементы)
		HTTP RESPONCE (ответ сервера)
			операция выполнена успешно 
				HTTP CODE=200
			ошибка в передаваемых аргументах
				HTTP CODE=400
			внутренняя ошибка сервера (а также другие ошибки для которых было принято решение не создавать детального описания в ответе)
				HTTP CODE=500
	
	добавить товар
		/item
		HTTP REQUEST / METHOD=POST
			m=POST
			title//название
			idCat//ид категории
			price//цена
		HTTP RESPONCE
			операция выполнена успешно 
				HTTP CODE=201
				id
			ошибка в передаваемых аргументах
				HTTP CODE=400
			внутренняя ошибка сервера (а также другие ошибки для которых было принято решение не создавать детального описания в ответе)
				HTTP CODE=500
	
	редактировать товар
		/item/id
			//id - ид элемента
		HTTP REQUEST / METHOD=POST
			m=PUT
			target=item
			title//название
			idCat//ид категории
			price//цена
		HTTP RESPONCE
			операция выполнена успешно 
				HTTP CODE=200
			указанный элемент не найден
				HTTP CODE=404
			ошибка в передаваемых аргументах
				HTTP CODE=400
			внутренняя ошибка сервера (а также другие ошибки для которых было принято решение не создавать детального описания в ответе)
				HTTP CODE=500
	
	удалить товар
		/item/id
			//id - ид элемента
		HTTP REQUEST / METHOD=POST
			m=DELETE
			target=item
		HTTP RESPONCE
			операция выполнена успешно 
				HTTP CODE=200
			указанный элемент не найден
				HTTP CODE=404
			ошибка в передаваемых аргументах
				HTTP CODE=400
			внутренняя ошибка сервера (а также другие ошибки для которых было принято решение не создавать детального описания в ответе)
				HTTP CODE=500
